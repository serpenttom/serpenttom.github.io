Bubble Sort

Build sort_bbb.py 

y= [6,8,3,2,7,4,5,1] 
N= len(y) 
for i in range(0,N):
    for j in range(0,N-i-1):
        if y[j+1] < y[j]:
            answer=y[j]
            y[j]=y[j+1]
            y[j+1]=answer

Cocktail Sort

Build sort_cct.py 

N=len(y)
a,b=0,N-1
for i in range(0,N):
    if i%2==0:
        for j in range(a,b-1):
            if y[j+1] < y[j]:
                answer=y[j]
                y[j]=y[j+1]
                y[j+1]=answer
        b=b-1
    else:
        for j in range(b,a-1,-1):
            if y[j+1] < y[j]:
                answer=y[j]
                y[j]=y[j+1]
                y[j+1]=answer
        a=a+1
return(y)
