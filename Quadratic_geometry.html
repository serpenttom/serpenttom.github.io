Ellipse

$$x^2+\frac{y^2}{4}=1$$ $$\left\{ \begin{align}x=\cos(t)\\ y=2\sin(t)\end{align}\right., t\in [0,2\pi]$$

Ellipse.py 

import numpy as np
import plotly.graph_objs as go
t=np.linspace(0,2*np.pi,121)
a=1
b=2
x=a*np.cos(t)
y=b*np.sin(t)
D=go.Scatter(x=x,y=y)
L=go.Layout(title='哭途',width=390,height=390,plot_bgcolor='white',
            xaxis={"range":[-3,3],'zerolinecolor':'black'},yaxis={"range":[-3,3],'zerolinecolor':'black'})
F=go.Figure(data=D,layout=L)
F.write_html('tuoyuan.html')

Sphere

$$x^2+y^2+z^2=1$$ $$ \left\{\begin{align} x=\cos(t)\sin(s)\\ y=\sin(t)\sin(s)\\ z=\cos(s) \end{align} \right., t\in[0,2\pi],s\in[0,\pi]$$

Sphere.py

import numpy as np
import plotly.graph_objs as go
num=35
T=np.linspace(0,2*np.pi,num)
Z=np.linspace(0,np.pi,num)
t,r=np.meshgrid(T,Z)
x=np.cos(t)*np.sin(r)
y=np.sin(t)*np.sin(r)
z=np.cos(r)
D1=go.Surface(x=x,y=y,z=z,surfacecolor=z,colorscale="blues")
F=go.Figure(data=[D1])
F.write_html('ball_3d_orth.html')

Hyperboloid of one sheet

$$x^2+y^2-z^2=1$$ $$ \left\{\begin{align} x=\cos(t)\cosh(s)\\ y=\sin(t)\cosh(s)\\ z=\sinh(s) \end{align} \right., t\in[0,2\pi],s\in R$$

Hyperboloid of one sheet.py

import numpy as np
import plotly.graph_objs as go
num=37
T=np.linspace(0,np.pi,num)
Z=np.linspace(-2,2,num)
t,z=np.meshgrid(T,Z)
sol=[2,1]
x=t*0
y=t*0
for i in range(num):
    for j in range(num):
        A=np.array([[-np.sin(t[i,j]),np.cos(t[i,j])],[np.cos(t[i,j]),np.sin(t[i,j])]])
        b=np.array([[np.cos(t[i,j])+z[i,j]*(-np.sin(t[i,j]))],[(-np.sin(t[i,j])-z[i,j]*np.cos(t[i,j]))]])
        sol=np.linalg.solve(A,b)
        x[i,j]=sol[0]
        y[i,j]=sol[1]
L=go.Layout(title='데秊崗혓충')
D=go.Surface(x=x,y=y,z=z,surfacecolor=t,colorscale="hsv")
F=go.Figure(data=D,layout=L)
F.write_html('d_shuangqu.html')
